const cta = document.getElementById('cta');

symbol.style.transition = 'none';

symbol.style.transform = 'translateY(0)';

```javascript

// ... código existente ...

function showResult(winningSymbol) {

result.textContent = `Parabéns! Você ganhou com ${winningSymbol}!`;

const jackpot = document.getElementById('jackpot');

jackpot.classList.remove('hidden');

const discountDiv = document.getElementById('discount');

const originalPrice = 100; // Insira o preço original do produto aqui

const discount = 0.4; // 40% de desconto

const discountedPrice = originalPrice * (1 - discount);

let currentPrice = originalPrice;

const countdown = setInterval(() => {

currentPrice -= discount * originalPrice / 100;

if (currentPrice <= discountedPrice) {

currentPrice = discountedPrice;

clearInterval(countdown);

}

discountDiv.innerHTML = `Preço original: R$${originalPrice.toFixed(2)}
Preço com desconto: R$${currentPrice.toFixed(2)}
Você será redirecionado em breve para adquirir o produto com o desconto ganho.`;

}, 30);

discountDiv.classList.remove('hidden');

setTimeout(() => {

window.location.href = 'https://www.example.com'; // Altere para o URL desejado

}, 3000);

}

Nesta versão atualizada do arquivo `script.js`, adicionamos uma função de intervalo chamada `countdown` que decrementa o preço atual a cada 30 milissegundos, criando uma animação de contador regressivo. A animação é interrompida quando o preço atual atinge o preço com desconto. O restante do código permanece inalterado.
